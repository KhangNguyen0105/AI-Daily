// Feed monitoring workflow:
export async function monitorProviderFeeds(): Promise<void> {
  // Run once per day, independent of collection pipeline
  const providers = await getAllProviders();
  
  for (const provider of providers) {
    // 1. Query provider API (if available)
    try {
      const apiModels = await provider.getProviderModelIds?.();
      for (const model of apiModels) {
        await recordDiscoveredModel(provider.name, model, 'api');
      }
    } catch (err) {
      // API not available, continue
    }
    
    // 2. Monitor changelog/RSS feeds
    const feedUrl = provider.getChangelogUrl?.(); // adapter-specific
    if (feedUrl) {
      const feedText = await fetchFeed(feedUrl);
      const foundModels = await detectNewModels(provider, feedText);
      for (const model of foundModels) {
        await recordDiscoveredModel(provider.name, model, 'feed');
      }
    }
  }
}

// Diff-based detection (runs after each crawl)
export async function detectChangesDiff(
  providerName: string,
  currentHtml: string,
): Promise<DiffResult> {
  // Fetch previous snapshot
  const previousRaw = await db.raw_data
    .select()
    .where({ source_id: provider.id })
    .orderBy({ crawledAt: 'desc' })
    .limit(2)
    .offset(1); // second-most-recent

  if (!previousRaw) return { new_models: [], renamed_models: [], deprecated_models: [], price_changes: [] };

  const previousHtml = previousRaw[0].evidence; // HTML snapshot
  return compareModelSnapshots(providerName, currentHtml, previousHtml);
}

// Compare snapshots: extract models from both, diff names/prices
export function compareModelSnapshots(
  providerName: string,
  currentHtml: string,
  previousHtml: string,
): DiffResult {
  // Parse both HTMLs for model lists
  const currentModels = parseModelsFromHtml(currentHtml); // provider-specific extraction
  const previousModels = parseModelsFromHtml(previousHtml);
  
  const currentNames = new Set(currentModels.map((m) => m.name));
  const previousNames = new Set(previousModels.map((m) => m.name));
  
  const newModels = [...currentNames].filter((name) => !previousNames.has(name));
  const deprecatedModels = [...previousNames].filter((name) => !currentNames.has(name));
  
  // Detect renames: same context_window, same family, different name
  const renamed = [];
  for (const prevModel of previousModels) {
    const renamedMatch = currentModels.find(
      (m) => m.family === prevModel.family &&
             m.context_window === prevModel.context_window &&
             m.name !== prevModel.name,
    );
    if (renamedMatch) {
      renamed.push({ old_name: prevModel.name, new_name: renamedMatch.name });
    }
  }
  
  // Detect price changes: >20% change
  const priceChanges = [];
  for (const prevModel of previousModels) {
    const currentMatch = currentModels.find((m) => m.name === prevModel.name);
    if (currentMatch && Math.abs(currentMatch.input_price - prevModel.input_price) / prevModel.input_price > 0.2) {
      priceChanges.push({
        model_name: prevModel.name,
        old_price: prevModel.input_price,
        new_price: currentMatch.input_price,
      });
    }
  }
  
  return { new_models: newModels, renamed_models: renamed, deprecated_models: deprecatedModels, price_changes: priceChanges };
}

export async function setupTier2Scheduler(connection: Redis): Promise<void> {
  const tier2Queue = new Queue('tier2-refresh', { connection });
  
  // Cron pattern for every 12 hours: 0 0,12 * * *
  // Runs at: 00:00 UTC and 12:00 UTC
  await tier2Queue.add(
    'refresh-tier2',
    { action: 'refresh_tier2_providers' },
    {
      repeat: {
        pattern: '0 0,12 * * *', // every 12 hours
        tz: 'UTC',
      },
      jobId: 'tier2-refresh-repeating',
      removeOnComplete: true,
    },
  );
  
  tier2Queue.process('refresh-tier2', async (job) => {
    const tier2Providers = await getAllTier2Providers();
    const collectQueue = new Queue('collect', { connection });
    
    for (const provider of tier2Providers) {
      await collectQueue.add(
        { providerName: provider.name, isScheduled: true },
        { priority: 1 }, // medium priority
      );
    }
    
    return { tier2ProvidersEnqueued: tier2Providers.length };
  });
}
