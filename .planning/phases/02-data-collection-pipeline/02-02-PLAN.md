```typescript
/**
 * Source tier classification for pricing data sources.
 * Tier 1: Official pricing page (highest trust)
 * Tier 2: Official blog/changelog/API docs
 * Tier 3: Third-party aggregators (lowest trust)
 */
export type SourceTier = 'tier1' | 'tier2' | 'tier3';

/**
 * A source URL with tier classification.
 * Each provider can have multiple sources at different tiers.
 */
export interface ProviderSource {
  url: string;
  tier: SourceTier;
  sourceType: 'pricing_page' | 'api_docs' | 'changelog' | 'blog';
}
```

**Create src/pipeline/confidence.ts** implementing the confidence scoring rules from RESEARCH.md:

The module exports three functions:
- `hasAllFields(e: ExtractionResult): boolean` -- returns true if modelName, inputPricePer1m, outputPricePer1m, and contextWindow are all non-null
- `hasCoreFields(e: ExtractionResult): boolean` -- returns true if modelName, inputPricePer1m, and outputPricePer1m are all non-null
- `calculateConfidence(tier: SourceTier, extraction: ExtractionResult, verificationPassed: boolean): 'verified' | 'likely' | 'low_confidence'`

Confidence rules (from research, per key decision 2):
- `verified`: tier === 'tier1' AND hasAllFields(extraction) AND verificationPassed === true
- `likely`: (tier === 'tier1' AND hasCoreFields(extraction)) OR (tier === 'tier2' AND hasAllFields(extraction) AND verificationPassed === true)
- `low_confidence`: everything else (tier3, or missing core fields, or failed verification)

Import ExtractionResult from '../providers/base' and SourceTier from '../providers/types'.

**Create src/pipeline/verification.ts:**

The module implements two-pass verification to detect LLM hallucinations (DCOL-07, per research Pattern 2).

**Exported types:**

```typescript
export interface VerificationModelResult {
  modelName: string;
  inputPricePer1m: number;
  outputPricePer1m: number;
  contextWindow: number;
  supported: boolean;  // Whether this data point is directly supported by source text
  evidenceQuote: string;  // Exact quote from source supporting this data point
}

export interface VerificationResult {
  verified: boolean;  // true if all pass1 results match pass2
  disagreements: Disagreement[];
  pass2Results: VerificationModelResult[];
}

export interface Disagreement {
  modelName: string;
  field: 'inputPricePer1m' | 'outputPricePer1m' | 'contextWindow';
  pass1Value: number;
  pass2Value: number;
  pass2Supported: boolean;
}
```

**Exported functions:**

1. `compareResults(pass1: ExtractionResult[], pass2: VerificationModelResult[]): Disagreement[]`
   - For each model in pass1, find matching model in pass2 by modelName (case-insensitive)
   - Compare inputPricePer1m, outputPricePer1m, contextWindow
   - If values differ by more than 0.1% relative tolerance, create a Disagreement entry
   - If a model exists in pass1 but not pass2, create a Disagreement for all fields
   - If pass2 marks a field as supported=false, that counts as a disagreement
   - Return array of Disagreement objects

2. `verifyExtraction(html: string, pass1Results: ExtractionResult[], apiKey: string): Promise<VerificationResult>`
   - Truncate HTML to 100K chars
   - Call generateObject with a verification-specific Zod schema:
     ```
     const verificationSchema = z.object({
       models: z.array(z.object({
         modelName: z.string(),
         inputPricePer1m: z.number(),
         outputPricePer1m: z.number(),
         contextWindow: z.number(),
         supported: z.boolean(),
         evidenceQuote: z.string(),
       }))
     });
     ```
   - Use model: createOpenAI({ apiKey })('gpt-4o') (same as extraction model)
   - Prompt (DIFFERENT from extraction prompt -- evidence-anchored verification):
     "You are a verification auditor. Given the following HTML source and previously extracted pricing data, verify EACH data point by finding the exact supporting quote in the source. If a data point cannot be verified from the source text, mark supported: false and set the value to the previously extracted value. Only include models that appear in the previously extracted data. Prices must be per 1M tokens in USD."
   - Include the pass1Results JSON in the prompt so pass2 knows what to verify
   - Call compareResults(pass1Results, object.models)
   - Return { verified: disagreements.length === 0, disagreements, pass2Results: object.models }
   - On error: throw (let BullMQ retry)
