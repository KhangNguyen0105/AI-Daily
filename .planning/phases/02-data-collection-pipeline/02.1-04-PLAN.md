export async function setupTier1Scheduler(
  connection: Redis,
): Promise<void> {
  // Main daily orchestrator (existing)
  await setupDailyScheduler(connection);
  
  // Tier 1 refresh jobs (NEW for Wave 4)
  const tier1Queue = new Queue('tier1-refresh', { connection });
  
  // Cron pattern for every 4 hours: 0 */4 * * *
  // Runs at: 00:00, 04:00, 08:00, 12:00, 16:00, 20:00 UTC
  await tier1Queue.add(
    'refresh-tier1',
    { action: 'refresh_tier1_providers' },
    {
      repeat: {
        pattern: '0 */4 * * *', // every 4 hours
        tz: 'UTC',
      },
      jobId: 'tier1-refresh-repeating',
      removeOnComplete: true,
    },
  );
  
  // Tier 1 refresh handler
  tier1Queue.process('refresh-tier1', async (job) => {
    const tier1Providers = await getAllTier1Providers();
    const collectQueue = new Queue('collect', { connection });
    
    for (const provider of tier1Providers) {
      await collectQueue.add(
        { providerName: provider.name, isScheduled: true },
        { priority: 2 }, // higher than routine collects
      );
    }
    
    return { tier1ProvidersEnqueued: tier1Providers.length };
  });
}

| Boundary | Description |
|----------|-------------|
| Provider API → Canonical Registry | Provider model IDs are trusted identity sources; validate against pricing page extraction |
| Tier 1 Adapter → Canonical Registry | Adapters must not fabricate provider model IDs; only store IDs returned by API or explicitly in config |
| Scheduler → Collect Queue | Tier 1 refresh scheduler prioritizes; must not starve other providers |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-15 | Tampering | Tier 1 provider model IDs | mitigate | model IDs stored as-is from provider API; validate against pricing page in verification |
| T-02-16 | Information Disclosure | Provider-specific pricing structures | accept | Each provider's pricing model documented in adapter code; complex pricing (tiered, regional) flagged as edge cases |
| T-02-17 | Denial of Service | Tier 1 refresh job frequency | mitigate | Every 4 hours = 6 jobs/day; retry with exponential backoff; max 3 attempts before alert |
| T-02-18 | Spoofing | Currency conversion for CNY providers | mitigate | Use daily exchange_rates table (updated daily from external API); validate FX rate freshness >24h = warning |
